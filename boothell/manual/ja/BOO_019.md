```javascript

  port: 6379,
  host: 'localhost'

  const data = redisClient.get('my-key');
  res.send(`Hello, ${data}!`);

  console.log(`Server started on port ${port}`);

```
### Step 2: Configuring the Database

Create a new file called `database.js` and add the following code:
```javascript
const pg = require('pg');

// Define the database connection settings
const dbConfig = {
  host: 'localhost',
  user: 'your_username',
  password: 'your_password',
  port: 5432,
  database: 'your_database'
};

// Establish a connection to the database
const dbClient = new pg.Pool(dbConfig);

// Define a simple query function
async function query(sql) {
  const result = await dbClient.query(sql);
  return result.rows;
}

module.exports = { query };
```
### Step 3: Setting up the Server

Create a new file called `server.js` and add the following code:
```javascript
const express = require('express');
const app = express();
const database = require('./database');

// Define a simple route handler
app.get('/data', async (req, res) => {
  const data = await database.query(`SELECT * FROM my_table`);
  res.send(data);

  console.log(`Server started on port ${port}`);
});
```
