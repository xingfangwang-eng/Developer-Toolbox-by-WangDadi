import threading
from pymemcache.client import Client

class DistributedCache:
    def __init__(self):
        self.cache = Client()
        self.lock = threading.Lock()

    def get(self, key):
        with self.lock:
            return self.cache.get(key)

    def set(self, key, value):
        with self.lock:
            self.cache.set(key, value)
