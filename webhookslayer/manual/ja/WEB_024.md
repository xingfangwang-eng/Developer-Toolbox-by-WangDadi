### 安装多线程优化

```go
package main

import (
	"fmt"
	"sync"
)

func main() {
	var wg sync.WaitGroup
	wg.Add(5)

	for i := 0; i < 5; i++ {
		go func(i int) {
			defer wg.Done()
			fmt.Println("Worker", i, "started")
			// do some work
			fmt.Println("Worker", i, "finished")
		}(i)
	}

	wg.Wait()
	fmt.Println("All workers finished")
}
```
