```
// Example: Automatic Sharding Indexing Implementation

using System;
using System.Collections.Generic;

class AutoShardingIndexExample
{
    private readonly Dictionary<int, List<string>> _shardedData = new Dictionary<int, List<string>>();

    public void ShardData(int key, string data)
    {
        // Shard the data based on the key
        int shardId = CalculateShardId(key);
        if (!_shardedData.ContainsKey(shardId))
        {
            _shardedData.Add(shardId, new List<string>());
        }
        _shardedData[shardId].Add(data);
    }

    private int CalculateShardId(int key)
    {
        // Calculate the shard ID based on the key
        return Math.Abs(key) % 10;
    }
}
```
